import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URL;

@RestController
@RequestMapping("/api/github")
public class GitHubFileController {

    /**
     * 读取GitHub仓库中的文件内容，通过完整URL指定
     *
     * @param githubUrl GitHub文件的完整URL
     * @param proxyHost 代理服务器地址（可选）
     * @param proxyPort 代理服务器端口（可选）
     * @return 文件内容或错误信息
     */
    @GetMapping("/read-file-by-url")
    public ResponseEntity<String> readFileByUrl(
            @RequestParam String githubUrl,
            @RequestParam(required = false) String proxyHost,
            @RequestParam(required = false) Integer proxyPort) {

        // 如果提供了代理配置，则设置系统代理
        if (proxyHost != null && proxyPort != null) {
            System.setProperty("http.proxyHost", proxyHost);
            System.setProperty("http.proxyPort", String.valueOf(proxyPort));
            System.setProperty("https.proxyHost", proxyHost);
            System.setProperty("https.proxyPort", String.valueOf(proxyPort));
        }

        try {
            String content = readGitHubFileViaAPI(githubUrl);
            return ResponseEntity.ok(content);
        } catch (Exception e) {
            return ResponseEntity.status(500).body("Error reading file: " + e.getMessage());
        }
    }

    /**
     * 使用GitHub API获取最新文件内容
     */
    private String readGitHubFileViaAPI(String githubUrl) throws Exception {
        try {
            // 从URL中解析出owner、repo和filePath
            String[] parts = githubUrl.split("/");
            if (parts.length < 7 || !parts[5].equals("blob")) {
                throw new Exception("Invalid GitHub URL format. Expected: https://github.com/{owner}/{repo}/blob/{branch}/{path}");
            }
            
            String owner = parts[3];
            String repo = parts[4];
            String branchAndPath = String.join("/", java.util.Arrays.copyOfRange(parts, 6, parts.length));
            String filePath = branchAndPath.replaceFirst("^[^/]+/", ""); // 移除分支名，保留文件路径
            
            // 获取最新commit SHA
            String commitsUrl = "https://api.github.com/repos/" + owner + "/" + repo + "/commits?path=" + filePath;
            String latestCommitSha = getLatestCommitSHA(commitsUrl);
            
            if (latestCommitSha == null || latestCommitSha.isEmpty()) {
                throw new Exception("Failed to retrieve commit information.");
            }
            
            // 使用commit SHA获取特定版本的文件
            String fileUrl = "https://raw.githubusercontent.com/" + owner + "/" + repo + "/" + latestCommitSha + "/" + filePath;
            return fetchFileContent(fileUrl);
            
        } catch (Exception e) {
            throw new Exception("Failed to read GitHub file: " + e.getMessage());
        }
    }
    
    private String getLatestCommitSHA(String apiUrl) throws IOException {
        HttpURLConnection connection = null;
        BufferedReader reader = null;
        
        try {
            URL url = new URL(apiUrl);
            connection = (HttpURLConnection) url.openConnection();
            connection.setRequestMethod("GET");
            connection.setRequestProperty("User-Agent", "Java-GitHub-Client");
            connection.setUseCaches(false);
            
            if (connection.getResponseCode() == 200) {
                reader = new BufferedReader(new InputStreamReader(connection.getInputStream()));
                String response = reader.readLine();
                // 简单解析JSON获取第一个commit的SHA
                int start = response.indexOf("\"sha\":\"") + 7;
                int end = response.indexOf("\"", start);
                if (start > 6 && end > start) {
                    return response.substring(start, end);
                }
            }
        } finally {
            if (reader != null) reader.close();
            if (connection != null) connection.disconnect();
        }
        return null;
    }
    
    private String fetchFileContent(String fileUrl) throws IOException {
        HttpURLConnection connection = null;
        BufferedReader reader = null;
        StringBuilder content = new StringBuilder();
        
        try {
            URL url = new URL(fileUrl);
            connection = (HttpURLConnection) url.openConnection();
            connection.setRequestMethod("GET");
            connection.setRequestProperty("User-Agent", "Java-GitHub-Client");
            connection.setUseCaches(false);
            connection.addRequestProperty("Cache-Control", "no-cache, no-store, must-revalidate");
            
            if (connection.getResponseCode() == 200) {
                reader = new BufferedReader(new InputStreamReader(connection.getInputStream()));
                String line;
                while ((line = reader.readLine()) != null) {
                    content.append(line).append("\n");
                }
                if (content.length() > 0) {
                    content.deleteCharAt(content.length() - 1);
                }
            } else {
                throw new IOException("HTTP Error: " + connection.getResponseCode());
            }
        } finally {
            if (reader != null) reader.close();
            if (connection != null) connection.disconnect();
        }
        
        return content.toString();
    }
}
